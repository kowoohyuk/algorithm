# 백준

## 바이러스

### 풀이

**코드**

```javascript
const readline = require("readline");

const rl = readline.createInterface({
  input: process.stdin,
  output: process.stdout,
});

let N = 0;
let length = 0;
const arr = [];

const solution = (N, length, arr) => {
  const graph = {};
  arr.forEach(([x, y]) => {
    if(graph[x]) {
      graph[x].push(y);
    } else {
      graph[x] = [y];
    }
  });
  const visited = new Set();
  const dfs = i => {
    if(!graph[i]) return;
    while(graph[i].length > 0) {
      const v = graph[i].pop();
      visited.add(v);
      dfs(v);
    }
  };
  dfs('1');
  console.log(visited.size);
};

rl.on("line", function (line) {
  if(N === 0) {
    N = +line;
    return;
  }
  if(length === 0) {
    length = +line;
    return;
  }
  arr.push(line.split(' '));
  if(length === arr.length) {
    solution(N, length, arr);
    rl.close();
  }
}).on("close", function () {
  process.exit();
});
```
